Adaptive Earnings Formula
// smartGrid Earnings Calculation function calculateEarnings(stream) { const baseRate = 0.001; // per viewer-minute const engagementMultiplier = (stream.avgWatchTime / 60) * 0.5 + (stream.revisitRate) * 0.3 + (stream.tipVelocity) * 0.2; const boost = Math.min(engagementMultiplier, 3.0); return baseRate * boost * stream.viewerMinutes; }
